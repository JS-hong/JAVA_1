create table RoomDB(
rno number primary key,
room_type varchar2(10),
room_price number,
max_value number
);

create table RoomHistoryDB(
rno number not null,
room_clean varchar2(10),
dns_check varchar2(10),
check_in varchar2(10),
CONSTRAINT fk_rno foreign key(rno) references RoomDB(rno)
);

create table CustomerDB(
cno varchar2(20) primary key,
cname varchar2(30), 
cphone varchar2(13), 
cemail varchar2(30),
in_date DATE,
out_date DATE,
rsv_total number,
rno varchar(10) foreign key
);
